import cmath

def resolver_equacao(a, b, c):
    if a == 0:
        return "O coeficiente 'a' deve ser diferente de zero."

    delta = (b**2) - (4*a*c)

    x1 = (-b + cmath.sqrt(delta)) / (2*a)
    x2 = (-b - cmath.sqrt(delta)) / (2*a)

    return delta, x1, x2

a = 1
b = -5
c = 6

print(f"Equação gerada: {a}x² + ({b})x + ({c}) = 0\n")

resultado = resolver_equacao(a, b, c)

if isinstance(resultado, tuple):
    delta, raiz1, raiz2 = resultado
    print(f"Valor de Delta: {delta}")
    
    if delta >= 0:
        print(f"Raízes reais: x1 = {raiz1.real:.2f}, x2 = {raiz2.real:.2f}")
    else:
        print(f"Raízes complexas: x1 = {raiz1:.2f}, x2 = {raiz2:.2f}")
else:
    print(resultado)
